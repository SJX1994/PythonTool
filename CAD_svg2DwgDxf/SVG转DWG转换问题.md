根据搜索结果，Python 确实有办法将带贝塞尔曲线（BCurve）的 SVG 文件转换为 DXF 格式。以下是几种解决方案：

## 方法一：使用 svgpathtools + ezdxf（推荐）

这是最常用的方法，可以将 SVG 中的贝塞尔曲线转换为 DXF 格式。

### 安装依赖库：

pip install svgpathtools ezdxf numpy


### 代码示例：

import svgpathtools
from svgpathtools import parse_path, Path
import ezdxf
from ezdxf.math import Bezier4P

# 1. 读取SVG文件
def svg_to_dxf(input_svg, output_dxf):
    # 解析SVG文件
    paths, attributes = svgpathtools.svg2paths(input_svg)
    
    # 创建DXF文档
    doc = ezdxf.new('R2010')  # 可选 R2010, R2007 等，适用于2018以下版本
    msp = doc.modelspace()
    
    # 遍历SVG中的所有路径
    for path in paths:
        # 将贝塞尔曲线离散化为多个点
        points = discretize_path(path, tolerance=0.5)
        
        if len(points) > 1:
            # 创建多段线（LWPOLYLINE）
            msp.add_lwpolyline(points)
    
    # 保存DXF文件
    doc.saveas(output_dxf)
    print(f"转换完成: {output_dxf}")

def discretize_path(path, tolerance=0.5):
    """将贝塞尔曲线离散化为点序列"""
    points = []
    
    for segment in path:
        # 获取曲线的起始点
        points.append((segment.start.real, segment.start.imag))
        
        # 对于贝塞尔曲线，需要采样多个点

def discretize_cubic_bezier(start, control1, control2, end, tolerance):
    """将三次贝塞尔曲线离散化为点"""
    points = []
    
    # 使用 svgpathtools 内置的长度计算
    length = abs(end - start) * 1.5  # 粗略估计
    
    # 根据曲线长度计算采样点数
    num_samples = max(2, int(length / tolerance))
    
    for i in range(num_samples + 1):
        t = i / num_samples
        # 三次贝塞尔公式
        t2 = t * t
        t3 = t2 * t
        mt = 1 - t
        mt2 = mt * mt
        mt3 = mt2 * mt
        
        x = (mt3 * start.real + 
             3 * mt2 * t * control1.real + 
             3 * mt * t2 * control2.real + 
             t3 * end.real)
        y = (mt3 * start.imag + 
             3 * mt2 * t * control1.imag + 
             3 * mt * t2 * control2.imag + 
             t3 * end.imag)
        
        points.append(complex(x, y))
    
    return points


def discretize_quadratic_bezier(start, control, end, tolerance):
    """将二次贝塞尔曲线离散化为点"""
    # 将二次贝塞尔转换为三次贝塞尔
    # 公式: CP1 = P0 + 2/3*(P1-P0), CP2 = P2 + 2/3*(P1-P2)
    
    cp1 = start + (control - start) * 2/3
    cp2 = end + (control - end) * 2/3
    
    return discretize_cubic_bezier(start, cp1, cp2, end, tolerance)

    end = arc.end

    end_angle = np.angle(end - arc.center())
    
    # 处理弧度方向
    if arc.arc_flag == 0:  # 小弧
        if end_angle < start_angle:
            end_angle += 2 * np.pi
    else:  # 大弧
        if end_angle > start_angle:
            end_angle -= 2 * np.pi
    
    # 计算弧长和采样点数
    angle_diff = abs(end_angle - start_angle)
    arc_length = radius * angle_diff
    num_samples = max(2, int(arc_length / tolerance))
    
    for i in range(num_samples + 1):
        t = i / num_samples
        angle = start_angle + t * (end_angle - start_angle)
        x = arc.center().real + radius * np.cos(angle)
        y = arc.center().imag + radius * np.sin(angle)
        points.append(complex(x, y))
    
    return points

def discretize_cubic_bezier(start, control1, control2, end, tolerance):
    """将三次贝塞尔曲线离散化为点"""
    points = []
    
    # 使用更准确的采样
    length = abs(end - start)
    num_samples = max(3, int(length / tolerance) + 1)
    
    for i in range(num_samples + 1):
        t = i / num_samples
        t2 = t * t
        t3 = t2 * t
        mt = 1 - t
        mt2 = mt * mt
        mt3 = mt2 * mt
        
        x = (mt3 * start.real + 
             3 * mt2 * t * control1.real + 
             3 * mt * t2 * control2.real + 
             t3 * end.real)
        y = (mt3 * start.imag + 
             3 * mt2 * t * control1.imag + 
             3 * mt * t2 * control2.imag + 
             t3 * end.imag)
        
        points.append(complex(x, y))
    
    return points


def discretize_quadratic_bezier(start, control, end, tolerance):
    """将二次贝塞尔曲线离散化为点"""
    cp1 = start + (control - start) * 2/3
    cp2 = end + (control - end) * 2/3
    
    return discretize_cubic_bezier(start, cp1, cp2, end, tolerance)

    end = arc.end
    radius = abs(arc.radius)

    end_angle = np.angle(end - arc.center())

    else:
        if end_angle > start_angle:
            end_angle -= 2 * np.pi

        x = arc.center().real + radius * np.cos(angle)
        y = arc.center().imag + radius * np.sin(angle)
        points.append(complex(x, y))
    
    return points

def discretize_cubic_bezier(start, control1, control2, end, tolerance):
    """将三次贝塞尔曲线离散化为点"""
    points = []
    
    length = abs(end - start)
    num_samples = max(3, int(length / tolerance) + 1)
    
    for i in range(num_samples + 1):
        t = i / num_samples
        t2 = t * t
        t3 = t2 * t
        mt = 1 - t
        mt2 = mt * mt
        mt3 = mt2 * mt
        
        x = (mt3 * start.real + 
             3 * mt2 * t * control1.real + 
             3 * mt * t2 * control2.real + 
             t3 * end.real)
        y = (mt3 * start.imag + 
             3 * mt2 * t * control1.imag + 
             3 * mt * t2 * control2.imag + 
             t3 * end.imag)
        
        points.append(complex(x, y))
    
    return points


def discretize_quadratic_bezier(start, control, end, tolerance):
    """将二次贝塞尔曲线离散化为点"""
    cp1 = start + (control - start) * 2/3
    cp2 = end + (control - end) * 2/3
    
    return discretize_cubic_bezier(start, cp1, cp2, end, tolerance)


def discretize_arc_fixed(arc, tolerance):
    """将圆弧离散化为点 - 修复版"""
    points = []
    
    start = arc.start
    end = arc.end
    
    # 修复：center 可能是属性而不是方法
    try:
        # 尝试作为属性访问
        if hasattr(arc, 'center') and callable(arc.center):
            center = arc.center()
        else:
            center = arc.center
    except:
        # 如果失败，手动计算圆心
        # 使用起点、终点和半径估算圆心
        radius = abs(arc.radius)
        mid = (start + end) / 2
        chord = end - start
        chord_len = abs(chord)
        
        if chord_len > 0 and radius > chord_len / 2:
            # 计算圆心到中点的距离
            h = np.sqrt(max(0, radius**2 - (chord_len/2)**2))
            # 垂直于弦的方向
            perp = complex(-chord.imag, chord.real) / chord_len
            center = mid + perp * h
        else:
            center = mid
    
    try:
        radius = abs(arc.radius)
        
        # 计算角度
        start_angle = np.angle(start - center)
        end_angle = np.angle(end - center)
        
        # 处理弧度方向
        if hasattr(arc, 'arc_flag'):
            if arc.arc_flag == 0:  # 小弧
                if end_angle < start_angle:
                    end_angle += 2 * np.pi
            else:  # 大弧
                if end_angle > start_angle:
                    end_angle -= 2 * np.pi
        
        angle_diff = abs(end_angle - start_angle)
        arc_length = radius * angle_diff
        num_samples = max(2, int(arc_length / tolerance))
        
        for i in range(num_samples + 1):
            t = i / num_samples
            angle = start_angle + t * (end_angle - start_angle)
            x = center.real + radius * np.cos(angle)
            y = center.imag + radius * np.sin(angle)
            points.append(complex(x, y))
            
    except Exception as e:
        # 如果圆弧处理失败，直接返回起点和终点
        points = [start, end]
    
    return points

def svg_to_dxf_by_mesh(input_svg, output_dxf, tolerance=2.0, gap_threshold=50, debug=True):
    """
    按Mesh面/独立区域分离SVG路径
    gap_threshold: 判断路径断开的距离阈值（DXF单位）
    """
    
    # 1. 读取SVG文件
    paths, attributes = svgpathtools.svg2paths(input_svg)
    
    if debug:
        print(f"发现 {len(paths)} 个路径对象")
    
    # 2. 计算全局边界和缩放
    all_coords = []
    for path in paths:
        for segment in path:
            all_coords.append((segment.start.real, segment.start.imag))
            all_coords.append((segment.end.real, segment.end.imag))
    
    if not all_coords:
        print("错误：未找到任何坐标点")
        return
    
    x_coords = [c[0] for c in all_coords]
    y_coords = [c[1] for c in all_coords]
    
    min_x, max_x = min(x_coords), max(x_coords)
    min_y, max_y = min(y_coords), max(y_coords)
    
    if debug:
        print(f"SVG边界: X=[{min_x}, {max_x}], Y=[{min_y}, {max_y}]")
    
    # 缩放
    svg_width = max_x - min_x
    svg_height = max_y - min_y
    scale = 200 / max(svg_width, svg_height)
    
    center_x = (min_x + max_x) / 2
    center_y = (min_y + max_y) / 2
    
    def transform_point(pt):
        x = (pt.real - center_x) * scale
        y = -(pt.imag - center_y) * scale
        return (round(x, 2), round(y, 2))
    
    # 3. 创建DXF文档
    doc = ezdxf.new('R2010')
    msp = doc.modelspace()
    
    # 存储所有独立的polyline组
    all_polylines = []
    polyline_count = 0
    
    # 4. 处理每个路径
    for path_idx, path in enumerate(paths):
        if debug:
            print(f"\n路径 {path_idx + 1}: {len(path)} 个段")
        
        # 当前组的点
        current_points = []
        prev_point_dxf = None
        
        # 遍历每个段
        for seg_idx, segment in enumerate(path):
            try:
                start_point = transform_point(segment.start)
                end_point = transform_point(segment.end)
                
                # 检查是否需要开始新的polyline（路径断开）
                if prev_point_dxf is not None:
                    # 计算与上一个点的距离
                    dist = abs(complex(end_point[0] - prev_point_dxf[0], 
                                      end_point[1] - prev_point_dxf[1]))
                    
                    # 如果距离超过阈值，说明是新的mesh面
                    if dist > gap_threshold:
                        # 保存当前的polyline
                        if len(current_points) > 1:
                            all_polylines.append(current_points)
                            current_points = []
                
                # 开始新组或继续当前组
                if len(current_points) == 0:
                    current_points.append(start_point)
                
                # 根据曲线类型添加点
                if isinstance(segment, Line):
                    current_points.append(end_point)
                    prev_point_dxf = end_point
                    
                elif isinstance(segment, CubicBezier):
                    points = discretize_cubic_bezier(
                        segment.start, segment.control1, 
                        segment.control2, segment.end, tolerance
                    )
                    for p in points[1:]:
                        pt = transform_point(p)
                        current_points.append(pt)
                    prev_point_dxf = current_points[-1]
                        
                elif isinstance(segment, QuadraticBezier):

                        current_points.append(pt)

            except Exception as e:

                    print(f"  段 {seg_idx + 1} 错误: {e}")
                continue
        
        # 保存最后一个polyline
        if len(current_points) > 1:
            all_polylines.append(current_points)
    
    # 5. 将所有polyline写入DXF
    if debug:
        print(f"\n共分离出 {len(all_polylines)} 个独立区域")

    doc.saveas(output_dxf)
    print(f"\n✓ 转换完成: {output_dxf}")
    print(f"  共创建 {polyline_count} 条多段线")

    for path in root.findall('{http://www.w3.org/2000/svg}path'):

def parse_transform(transform_str):
    """解析SVG transform属性"""
    if not transform_str:
        return None
    
    # 匹配 matrix(a b c d e f) 格式
    match = re.search(r'matrix\(([^)]+)\)', transform_str)
    if match:
        values = [float(x) for x in match.group(1).split()]
        # matrix(a, b, c, d, e, f) 对应:
        # | a c e |
        # | b d f |
        # | 0 0 1 |
        return {
            'a': values[0], 'b': values[1], 
            'c': values[2], 'd': values[3],
            'e': values[4], 'f': values[5]
        }
    
    return None


def apply_transform(pt, transform):
    """应用SVG变换矩阵到点"""
    if not transform:
        return pt
    
    # x' = a*x + c*y + e
    # y' = b*x + d*y + f
    x = transform['a'] * pt.real + transform['c'] * pt.imag + transform['e']
    y = transform['b'] * pt.real + transform['d'] * pt.imag + transform['f']
    
    return complex(x, y)


def discretize_cubic(control1, control2, end, tolerance):
    """离散化三次贝塞尔曲线"""
    points = []
    length = abs(end - control1) + abs(control2 - control1)
    num_samples = max(3, int(length / tolerance) + 1)
    
    start = complex(0, 0)  # 起点已在主函数中添加
    
    for i in range(1, num_samples + 1):
        t = i / num_samples
        t2 = t * t
        t3 = t2 * t
        mt = 1 - t
        mt2 = mt * mt
        mt3 = mt2 * mt
        
        x = mt3 * 0 + 3*mt2*t*control1.real + 3*mt*t2*control2.real + t3*end.real
        y = mt3 * 0 + 3*mt2*t*control1.imag + 3*mt*t2*control2.imag + t3*end.imag
        
        points.append(complex(x, y))
    
    return points
